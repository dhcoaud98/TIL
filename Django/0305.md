import random
import requests

# 1. url 주소를 따로 지정한다. 길이가 길기 대문에 반복하여 사용하면 코드 사용성이 낮아지기 때문이다.
url = 'https://...'

# 2. 요청, 응답
response = requests.get(url)
movie = response.json()

# 3. url 주소의 내용을 사용 목적에 맞게 가공하기
# url의 주소를 movie에 넣어서 사용함. movie는 dictionary 형태 이므로 그중에 key 값이 'bonus'이 value를 win에 할당 한다는 뜻이다. 
win = movie['bonus']
