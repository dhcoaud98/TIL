```python
#재귀 : 무한 돌지 않기 위해 base case를 생각하기!!

def sum_of_digit(number):
    # base case
    if number < 0 :
        return number
    else:
        number, remainder = divmod(number, 10)
        return sum_of_digit(number) + remainder
```
